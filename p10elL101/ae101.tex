	\begin{enumerate}[1.]
		
		%Task 1
		{\bf \item The following devices are connected (in parallel) in a common circuit of the $\SI{230}{\volt}$ network, which is protected by $\SI{10}{\ampere}$:} \\
		$G_1$ with $ \SI{75}{\watt};\quad G_2$ with $\SI{1.5}{\ampere}; \quad G_3 $ with $\SI{45}{\ohm}$.
		
		\begin{enumerate}[a)]
			\item Calculate the following electrical quantities of the connected devices:
			$$ I_1=\dots \quad P_2=\dots \quad I_3=\dots$$
			\item Calculate the maximum power of another device $G_4$ that can also be connected in parallel in this circuit.
		\end{enumerate}
		
		%Exercise 2
		{\bf \item A current $I = \SI{0.18}{\ampere}$ flows through two electrical resistors $R_1$ and $R_2$.} A voltage of $U_1=\SI{9,0}{\volt}$ falls across $R_1$, the size of $R_2$ is
		$\SI{150}{\ohm}$.\\
		Calculate:
		$$R_1=\dots \quad U_2=\dots \quad U_{ges}=\dots $$
		
		%Task 3
		{\bf \item An electrical device with ($\SI{110}{\volt} / \SI{0.4}{\ampere} $) should be connected to the voltage $\SI{230}{\volt}$ \-closed.} Calculate the size of the necessary electrical series resistance $R_{vor}$.
		
		%Task 4
		{\bf \item Show with a short calculation how the electrical resistance $R$ of a metal wire changes when $\dots$}
		\begin{enumerate}[a)]
			\item $\dots$the wire length is doubled and at the same time the cross-sectional area is tripled.
			\item $\dots$the wire length is quadrupled and the radius is doubled
		\end{enumerate}
		
		%Task 5
		{\bf \item A 17.5 m long copper pair cable
			$\bigl( \rho_{Cu}= \SI{0.017}{\ohm\cdot\milli\meter^2 \per \meter}\bigr)$
			has an electrical resistance of $R=\SI{0.20}{\ohm}$.}\\
		What is the cross-sectional area and radius of the wire used?
		
		{\bf Mark all true statements in tasks 6 and 7!}
		
		%Task 6
		{\bf \item The following applies to a pure parallel connection of several electrical resistors always}:\\
		
		With each additional resistor connected, $\dots$\\
		
		\begin{tabular}{p{75 pt}p{75 pt}p{75 pt}p{75 pt}}
			$\square R_{tot}$ larger& $\square R_{tot}$ smaller&
			$\square I_{ges}$ larger & $\square I_{ges}$ smaller
		\end{tabular}
		
		%Task 7
	{\bf	\item If two resistors $R_1$ and $R_2$ are connected in parallel, where $R_1>R_2$ applies, then  always} $\dots$ applies:
		\begin{enumerate}
			\item $\dots$ for $R_{ges}$:
			
			\begin{tabular}{p{75 pt}p{75 pt}p{75 pt}p{75 pt}}
				$\square R_{ges}>R_1 $&$ \square R_{ges}<R_1 $&$
				\square R_{ges}>R_2 $&$ \square R_{ges}<R_2 $
			\end{tabular}
			
			\item $\dots$ for the currents $I_1$ and $I_2$:
			
			\begin{tabular}{p{100 pt}p{100 pt}p{100 pt}}
				$\square I_1>I_2 $ & $ \square I_1=I_2 $ & $ \square I_1 <I_2 $
			\end{tabular}
			
			\item $\dots$ for the voltages $U_1$ and $U_2$:
			
			\begin{tabular}{p{100 pt}p{100 pt}p{100 pt}}
				$\square U_1>U_2 $ & $ \square U_1=U_2 $ & $ \square U_1 <U_2 $
			\end{tabular}
		\end{enumerate}
		
		%Task 8
		{\bf \item Which of the lights }
		$$L_1 ( \SI{4,0}{\volt} / \SI{0,1}{\ampere}); \quad L_2 (\SI{12}{\volt} / \SI{1,2}{\watt})
		~\textnormal{and} \quad L_3 (\SI{4,0}{\volt} / \SI{20}{\ohm})$$
		can be physically connected in a series connection?
		\begin{enumerate}[a)]
			\item Determine the necessary physical quantities from the data and decide!
			\item Calculate the operating current $I_{ges}$, the total resistance $R_{ges}$, and the total voltage $U_{ges}$ for the possible lamps from 8.a).
		\end{enumerate}
