	\noindent{\bf Task 3. b)}\\
	$$I= {Q \over t}={\SI{4,0}{\ampere\hour} \over \SI{48}{\hour}}=\SI{83}{\milli\ampere}$$

		\noindent{\bf Task 3. c)}\\
		12 minutes is ${1\over 5}\SI{}{\hour}$:
		$$Q=I\cdot t = \SI{0.020}{\ampere} \cdot {1\over 5}\SI{}{\hour} = \SI{4,0}{\milli\ampere\hour} $$
		\vspace{0.5 cm}
		
		\noindent{\bf Task 3. d)}\\
		The operating current is:
		$$I={P\over U}={\SI{6000}{\watt}\over \SI{60}{\volt}}=\SI{100}{\ampere}$$
		This results in the capacity:
		$$Q=I\cdot t = \SI{100}{\ampere} \cdot \SI{1}{\hour} = \SI{100}{\ampere\hour}$$
