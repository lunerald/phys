	\section*{Formula of the day}
	
	\[
	F = m \cdot g
	\]
	\begin{itemize}
		\item \( F \): Weight in Newtons (N)
		\item \( m \): Mass in kilograms (kg)
		\item \( g \): Location factor / acceleration due to gravity in \( \frac{\text{m}}{\text{s}^2} \)
	\end{itemize}
